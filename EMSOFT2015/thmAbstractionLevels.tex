\begin{thm}
	Let $(\Sys,\Vc,\Pc,\{\Sys_i\})$ be a HCHA.
	Let $x \in \Sc_\Sys$ be a solution of CHA $\Sys$.
	Then for every $1 \leq i \leq a$, 
	there exists a solution $x^{(i)} \in \Sc_{\Sys_i}$ such that $x^{(i)} = \proj{x}{V_i}$.	
\end{thm}
